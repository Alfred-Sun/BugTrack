CREATE TABLE EXPERTS(
 EXPERTID VARCHAR(10),
 PRODID VARCHAR(10),
 PRIMARY KEY(EXPERTID));

CREATE TABLE BUG(
 BUGID VARCHAR(10),
 AUTHORID VARCHAR(10),
 AUTHOR VARCHAR(20),
 PRODID VARCHAR(10),
 PRODNAME VARCHAR(25),
 ENV VARCHAR(15),
 TYPE VARCHAR(15),
 DESCRIPTION VARCHAR(100),
 REP_DATE DATE,
 PRIORITY VARCHAR(10),
 STATUS VARCHAR(15),
 ASSIGN_TO VARCHAR(10),
 PRIMARY KEY(BUGID));

CREATE TABLE SOLUTIONS(
 BUGID VARCHAR(10),
 EXPERTID VARCHAR(10),
 EXPERT VARCHAR(20),
 SOLN VARCHAR(200),
 SOLN_DATE DATE,
 PRIMARY KEY(BUGID, EXPERTID));
